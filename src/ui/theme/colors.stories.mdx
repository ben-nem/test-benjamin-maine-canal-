import { filter, includes } from 'lodash';
import { Meta, ColorPalette, ColorItem } from '@storybook/addon-docs';
import theme from '../../styles/theme';

<Meta title='Theme/Colors' />

# Colors

<ColorPalette>
  <ColorItem
    title='primary'
    subtitle="theme's colors"
    colors={filter(
      theme.colors.primary,
      (_, colorName) =>
        !includes(colorName, 'Light') && !includes(colorName, 'Dark')
    )}
  />
  <ColorItem
    title='primary light'
    subtitle="theme's colors"
    colors={filter(theme.colors.primary, (_, colorName) =>
      includes(colorName, 'Light')
    )}
  />
  <ColorItem
    title='primary dark'
    subtitle="theme's colors"
    colors={filter(theme.colors.primary, (_, colorName) =>
      includes(colorName, 'Dark')
    )}
  />
</ColorPalette>
